# 5
movies = {
    "": [2.01, 4.59, 7.99, 11.83, 16],
    "": [3.19, 5.08, 6.73, 8.10, 9.35],
    "": [4.07, 6.92, 9.30, 11.29, 13.03],
    "": [2.72, 3.79, 4.45, 4.83, 5.11]}

# 
plt.figure(figsize=(20, 8))

# x
ind = np.arange(len(movies))

# 
width = 0.8
bin_width = width / 5

# DataFrame
movie_pd = pd.DataFrame(movies)


# index[0,1,2,3,4]5
for index in movie_pd.index:
    # ny
    day_tickets = movie_pd.iloc[index]

    # nx
    xs = ind - (bin_width * (2 - index))

    # label
    plt.bar(xs, day_tickets, width=bin_width, label="{}".format(index + 1))

    # 
    for ticket, x in zip(day_tickets, xs):
        plt.annotate(ticket, xy=(x, ticket), xytext=(x-0.1, ticket+0.1))

# 
plt.legend()
plt.ylabel("")
plt.title("5")
plt.grid(True)

# x
plt.xticks(ind, movie_pd.columns)
